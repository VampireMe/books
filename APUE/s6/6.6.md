# 6.6 系统标识
***

&emsp;&emsp;
uname 函数可以返回和当前主机和操作系统相关信息：

    #include <sys/utsname.h>
    int uname(struct utsname *buf);
    struct utsname {
        char sysname[];
        char nodename[];
        char release[];
        char version[];
        char machine[];
    #ifdef _GNU_SOURCE
        char domainname[];
    #endif
    };

&emsp;&emsp;
需要注意 nodename 不能用于引用网络通信主机，仅仅适用于引用 UUCP 网络上的主机。
如果需要返回 TCP 网络主机可以使用 gethostname 这个函数：

    #include <unistd.h>
    int gethostname(char* name, int namelen);
    #include <sys/utsname.h>
    #include <unistd.h>
    #include <cstdio>
    int main() {
        struct utsname buf;
        uname(&buf);
        printf("sysname:%s\n"
               "nodename:%s\n"
               "release:%s\n"
               "version:%s\n"
               "machine:%s\n"
               "domainname:%s\n",
               buf.sysname,buf.nodename,
               buf.release,buf.version,
               buf.machine,buf.domainname);
        char host[128];
        gethostname(host, sizeof(host));
        printf("hostname:%s\n", host);
        return 0;
    }
    
&emsp;&emsp;

    release:2.6.23.1-42.fc8
