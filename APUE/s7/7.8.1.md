# 7.8.1 控制进程和 CPU 亲和性
***

    #include <sched.h>
    #include <unistd.h>
    #include <pthread.h>
    void* (void *arg) {
        int a = 0;
        while(1) {
            a++;
        }
        return NULL;
    }
    
    int main() {
        cpu_set_t cpu_set;
        CPU_ZERO(&cpu_set);
        CPU_SET(1, &cpu_set); // 指定在 CPU#1 和 #3 上运行...
        CPU_SET(3, &cpu_set);
        sched_setaffinity(getpid(), sizeof(cpu_set), &cpu_set);
        pthread_t thread[5];
        for(int i = 0; i < 5; i++){
            pthread_create(thread + i, NULL, ok, NULL);
        }
        for(int i = 0; i < 5; i++) {
            pthread_join(thread[i], NULL);
        }
        return 0;
    }
